Usage Example

// Initialize the library with your API key
CoolhandJS.init('your-api-key-here');

// Attach feedback widget to any element
const element = document.getElementById('myContent');
CoolhandJS.attach(element, {
  onSuccess: (feedback, response) => {
    console.log('Feedback submitted:', feedback);
  },
  onError: (error) => {
    console.error('Error submitting feedback:', error);
  }
});
